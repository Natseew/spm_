// Approve or reject the WFH request function handleApproval(staff_id, req_date, isApproved) { fetch('/api/wfh_approval', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ staff_id: staff_id, req_date: req_date, approved: isApproved, rejected: !isApproved, }), }) .then(response => response.json()) .then(data => { console.log('Success:', data); }) .catch((error) => { console.error('Error:', error); }); }
